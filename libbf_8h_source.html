305  terminated string or NULL if memory error. *plen contains its

310 #define BF_FTOA_FORMAT_MASK (3 << 16)
311 
312 /* fixed format: prec significant digits rounded with (flags &
313  BF_RND_MASK). Exponential notation is used if too many zeros are
314  needed.*/
315 #define BF_FTOA_FORMAT_FIXED (0 << 16)
316 /* fractional format: prec digits after the decimal point rounded with
317  (flags & BF_RND_MASK) */
318 #define BF_FTOA_FORMAT_FRAC (1 << 16)
319 /* free format: use as many digits as necessary so that bf_atof()
320  return the same number when using precision 'prec', rounding to
321  nearest and the subnormal+exponent configuration of 'flags'. The
322  result is meaningful only if 'a' is already rounded to the wanted
323  precision.
324 
325  Infinite precision (BF_PREC_INF) is supported when the radix is a
326  power of two. */
327 #define BF_FTOA_FORMAT_FREE (2 << 16)
328 /* same as BF_FTOA_FORMAT_FREE but uses the minimum number of digits
329  (takes more computation time). */
330 #define BF_FTOA_FORMAT_FREE_MIN (3 << 16)
331 
332 /* force exponential notation for fixed or free format */
333 #define BF_FTOA_FORCE_EXP (1 << 20)
334 /* add 0x prefix for base 16, 0o prefix for base 8 or 0b prefix for
335  base 2 if non zero value */
336 #define BF_FTOA_ADD_PREFIX (1 << 21)
337 /* return "Infinity" instead of "Inf" and add a "+" for positive
338  exponents */
